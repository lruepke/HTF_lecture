xc      = 0.5 #collocation point
Ri      = c*((1-a2)+2*a2*xc) - K*2*a2
a2      = solve(Ri, a2)

Ri          = c*((1-a2)+2*a2*x) - K*2*a2
Wi          = diff(Ri,a2)
a2          = solve(integrate(Wi*Ri,(x,0,1)),a2)
